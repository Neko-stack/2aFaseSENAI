-- üìò Cria e seleciona o banco
CREATE DATABASE livraria;
USE livraria;

-- üë§ Clientes
CREATE TABLE cliente(
  id_cliente INT PRIMARY KEY AUTO_INCREMENT,
  nome VARCHAR(100),
  rg CHAR(11),
  cpf CHAR(11),
  bairro VARCHAR(45),
  rua VARCHAR(45)
);

-- üè¢ Editoras
CREATE TABLE editora(
  id_editora INT PRIMARY KEY AUTO_INCREMENT,
  nome_fantasia VARCHAR(100),
  razao_social VARCHAR(100),
  bairro VARCHAR(45),
  rua VARCHAR(45),
  telefone VARCHAR(20)
);

-- ‚úçÔ∏è Autores
CREATE TABLE autor(
  id_autor INT PRIMARY KEY AUTO_INCREMENT,
  nome VARCHAR(100),
  rg CHAR(11)
);

-- üí∞ Financiadores (quem banca os livros)
CREATE TABLE financiador(
  id_financiador INT PRIMARY KEY AUTO_INCREMENT,
  nome VARCHAR(100),
  cnpj CHAR(14)
);

-- üìñ Livros com suas chaves estrangeiras
CREATE TABLE livro(
  id_livro INT PRIMARY KEY AUTO_INCREMENT,
  isbn CHAR(13),
  titulo VARCHAR(100),
  preco DECIMAL(8,2),
  categoria VARCHAR(45),
  id_editora INT,
  id_autor INT,
  id_financiador INT,
  FOREIGN KEY (id_editora) REFERENCES editora(id_editora),
  FOREIGN KEY (id_autor) REFERENCES autor(id_autor),
  FOREIGN KEY (id_financiador) REFERENCES financiador(id_financiador)
);

-- üßæ Pedidos (cliente + livro + quantidade)
CREATE TABLE pedido(
  id_pedido INT PRIMARY KEY AUTO_INCREMENT,
  id_livro INT,
  FOREIGN KEY (id_livro) REFERENCES livro(id_livro),
  quantidade INT,
  id_cliente INT,
  FOREIGN KEY (id_cliente) REFERENCES cliente(id_cliente)
);
